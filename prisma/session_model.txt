model Session {
  id         String   @id @default(uuid())
  userId     String
  user       User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  token      String   @unique
  device     String
  userAgent  String?
  expiresAt  DateTime
  createdAt  DateTime @default(now())
  accessedAt DateTime @default(now())

  @@map("sessions")
}
